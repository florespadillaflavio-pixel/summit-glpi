<app-form-field>
    <app-label>Número par</app-label>
    <input appInput type="number" placeholder="0" [formControl]="numeroPar">
    <button appSuffix appIconButton color="danger" *ngIf="numeroPar.value"
        (click)="numeroPar.setValue(null)"
    >
        <app-icon>clear</app-icon>
    </button>
    
    <app-hint>Escriba un número par </app-hint>
    <app-error *ngIf="numeroPar.hasError('noEsPar')">{{ numeroPar.value }} no es par</app-error>
    <app-success>{{ numeroPar.value }} es par</app-success>
</app-form-field>

import { Component } from '@angular/core';
import { FormControl, Validators } from '@angular/forms';

@Component({...})
export class MyComponent {

    readonly numeroPar = new FormControl(null, [
        Validators.required,
        (control) => {
            if (control.value === null) return null;
            if (control.value % 2 === 0) return null;
            else return { noEsPar: true };
        }
    ]);

}
