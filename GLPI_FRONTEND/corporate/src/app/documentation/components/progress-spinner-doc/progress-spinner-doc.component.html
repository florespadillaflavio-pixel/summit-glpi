Determinate Spinner Progress Spinner 
<app-spinner mode="determinate" [value]="percentage.value" />

<app-form-field>
    <app-label>Porcentaje</app-label>
    <button appPrefix appIconButton variant="tertiary"
        [disabled]="+percentage.value < 5"
        (click)="percentage.setValue(+percentage.value - 5)"
    >
        <app-icon>remove</app-icon>
    </button>
    <input appInput textAlign="center" min="0" max="100" [formControl]="percentage">
    <button appSuffix appIconButton variant="tertiary"
        [disabled]="+percentage.value > 95"
        (click)="percentage.setValue(+percentage.value + 5)"
    >
        <app-icon>add</app-icon>
    </button>
</app-form-field>

import { Component } from '@angular/core';
import { FormControl, Validators } from '@angular/forms';

@Component({...})
export class MyComponent {
    percentage = new FormControl(45, [Validators.min(0), Validators.max(100)]);
}

import { Component } from '@angular/core';
import { FormControl, Validators } from '@angular/forms';

@Component({...})
export class MyComponent {
    diameter = new FormControl(0, [Validators.min(0), Validators.max(100)]);
    strokeWidth = new FormControl(0, [Validators.min(0), Validators.max(50)]);
}
Diámetro Grosor de línea info INFO: El grosor de línea '[strokeWidth]' con un valor de 0 o null, se reconfigura al 10% del diámetro. Ambos valores están dados en 'px'.
