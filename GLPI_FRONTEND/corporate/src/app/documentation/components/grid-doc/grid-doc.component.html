Inputs

  • [max]: string | number = '1fr'
  • [gap]: string | number = '16px'
  • [noGap]: boolean = false
  • [template]: string | number
  • [justify]: 'center' | 'start' | 'end' | 'normal' = 'normal'
  • [align]: 'center' | 'start' | 'end' | 'stretch' | 'baseline' | 'normal' = 'normal'
